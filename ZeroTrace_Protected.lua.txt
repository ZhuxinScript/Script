local function _dec(data)
    local r = ""
    for i=1, #data do
        local b = data[i]
        local success, val = pcall(function() return bit32.bxor(b, _k) end)
        if not success then
            success, val = pcall(function() return b ^ _k end)
        end
        r = r .. ch(success and val or (b % 256))
    end
    return r
end
